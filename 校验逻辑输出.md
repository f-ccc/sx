## 一、选课记录表结构定义（等效 SQL 结构）

由于本系统使用 C 语言实现（非 SQL 数据库），以下为等效的
表结构定义，含字段约束和联合唯一索引。

### 1.1 Record 结构体（record.h）

```c
/* ↓↓↓ 等效于 SQL 表结构 ↓↓↓ */

typedef struct {
    char student_id[14];   /* VARCHAR(12) NOT NULL,      学号 */
    char name[21];         /* VARCHAR(20) NOT NULL,      姓名 */
    char college[41];      /* VARCHAR(40) NOT NULL,      学院 */
    char course_id[9];     /* VARCHAR(8)  NOT NULL,      课程编号 */
    char course_name[51];  /* VARCHAR(50) NOT NULL,      课程名称 */
    float credit;          /* DECIMAL(3,1) NOT NULL,     学分 */
    char semester[8];      /* VARCHAR(7)  NOT NULL,      选课学期 "YYYY-01/02" */
    Date enroll_date;      /* DATE NOT NULL,             选课日期 */
    int  score;            /* INT CHECK(0-100),          成绩 */
} Record;

/* 
 * 等效 SQL 约束：
 *   student_id  VARCHAR(12) NOT NULL  -- 12位纯数字
 *   course_id   VARCHAR(8)  NOT NULL  -- 8位课程编码
 *   semester    VARCHAR(7)  NOT NULL  -- 格式 YYYY-01/02
 *   enroll_date DATE NOT NULL         -- 须在选课时段内
 *   score       INT CHECK(score >= 0 AND score <= 100)
 *   credit      DECIMAL(3,1) CHECK(credit > 0 AND credit <= 20.0)
 *
 * 联合唯一索引（拦截同一学期重复选课）：
 *   UNIQUE INDEX idx_unique_enrollment 
 *     ON records(student_id, course_id, semester);
 *
 * 联合唯一索引（拦截同一学生重复选课）：
 *   UNIQUE INDEX idx_unique_student_course
 *     ON records(student_id, course_id);
 *   → 业务层区分：同学期=拦截，不同学期=重修放行
 */
```

## 二、重复选课判断核心代码（record.c）

```c
/* 
 * check_duplicate_triple_key()
 * ────────────────────────────
 * 功能：三重键（学号+课程编号+学期）完全一致判定为重复
 * 场景：普通正常选课 → 同一学期不可重复选课
 *       重修场景     → 不同学期/批次视为合法，不拦截
 * 返回：1=重复拦截, 0=放行
 *
 * 等效 SQL: SELECT COUNT(*) FROM records 
 *           WHERE student_id=? AND course_id=? AND semester=?
 */
int check_duplicate_triple_key(const Record *records, int count,
                                const char *student_id,
                                const char *course_id,
                                const char *semester)
{
    int i;

    if (records == NULL || student_id == NULL || 
        course_id == NULL || semester == NULL) {
        return 0;  /* 参数异常，放行（保守策略） */
    }

    for (i = 0; i < count; i++) {
        if (strcmp(records[i].student_id, student_id) == 0 &&
            strcmp(records[i].course_id, course_id) == 0 &&
            strcmp(records[i].semester, semester) == 0) {
            return 1;  /* ← 命中！同一学期重复选课，拦截 */
        }
    }

    return 0;  /* 不重复 或 不同学期（重修场景自动放行） */
}
```

## 三、字段合法性校验代码（record.c）

### 3.1 学号校验

```c
/*
 * 学号校验规则：
 *   ① 非空（由 validate_not_empty 前置拦截）
 *   ② 仅允许 0-9 数字字符，剔除字母/符号/空格
 *   ③ 固定 12 位长度
 * 等效 SQL: CHECK(student_id ~ '^[0-9]{12}$')
 */
int validate_student_id(const char *student_id)
{
    int len;
    int i;

    if (student_id == NULL) return INVALID_ID;

    len = strlen(student_id);
    if (len != 12) {
        return INVALID_ID;  /* 长度不符 */
    }

    for (i = 0; i < len; i++) {
        if (student_id[i] < '0' || student_id[i] > '9') {
            return INVALID_ID;  /* 含非法字符 */
        }
    }

    return OK;  /* 校验通过 */
}
```

### 3.2 选课日期校验

```c
/*
 * 选课日期校验规则：
 *   ① 非空，标准日期格式
 *   ② 日期本身合法（考虑闰年、各月天数）
 *   ③ 选课日期 ≥ 当期开学时间
 *      - 春季学期(01)：2月1日 ~ 4月30日
 *      - 秋季学期(02)：8月1日 ~ 10月31日
 *   ④ 选课日期 ≤ 当期选课截止时间
 *   ⑤ 禁止大于系统当前时间的未来无效日期
 *   ⑥ 重修场景：不受常规选课时间限制
 *      （目前简化：重修也需在合理日期范围内）
 * 等效 SQL: CHECK(enroll_date BETWEEN 
 *           CASE WHEN semester LIKE '%-01' 
 *                THEN DATE(CONCAT(LEFT(semester,4),'-02-01'))
 *                ELSE DATE(CONCAT(LEFT(semester,4),'-08-01'))
 *           END 
 *           AND CURRENT_DATE)
 */
int validate_enroll_date(const Date *date, const char *semester)
{
    int sem_year;
    int earliest_year, earliest_month, earliest_day;
    int latest_year, latest_month, latest_day;
    Date earliest, latest, now_estimate;
    time_t raw_time;
    struct tm *time_info;

    if (date == NULL) return INVALID_DATE;

    /* ① 基本日期合法性 */
    if (date->year < 2000 || date->year > 2099) return INVALID_DATE;
    if (date->month < 1 || date->month > 12)    return INVALID_DATE;
    if (date->day < 1 || 
        date->day > days_in_month(date->year, date->month)) {
        return INVALID_DATE;
    }

    /* ② 日期与学期时段匹配 */
    if (semester != NULL && strlen(semester) >= 7) {
        sem_year = (semester[0]-'0')*1000 + (semester[1]-'0')*100 +
                   (semester[2]-'0')*10  + (semester[3]-'0');

        if (semester[6] == '1') {  /* 春季 */
            earliest_month = 2;  earliest_day = 1;
            latest_month   = 4;  latest_day   = 30;
        } else {                    /* 秋季 */
            earliest_month = 8;  earliest_day = 1;
            latest_month   = 10; latest_day   = 31;
        }
        earliest_year = sem_year;
        latest_year   = sem_year;

        earliest.year = earliest_year;  earliest.month = earliest_month;  earliest.day = earliest_day;
        latest.year   = latest_year;    latest.month   = latest_month;    latest.day   = latest_day;

        if (date_compare(date, &earliest) == CMP_LESS) return INVALID_DATE;  /* 早于开学 */
        if (date_compare(date, &latest)   == CMP_GREATER) return INVALID_DATE;  /* 晚于截止 */
    }

    /* ③ 禁止未来日期 */
    raw_time = time(NULL);
    time_info = localtime(&raw_time);
    if (time_info != NULL) {
        now_estimate.year  = time_info->tm_year + 1900;
        now_estimate.month = time_info->tm_mon + 1;
        now_estimate.day   = time_info->tm_mday;
        if (date_compare(date, &now_estimate) == CMP_GREATER) {
            return INVALID_DATE;  /* 未来日期 */
        }
    }

    return OK;
}
```

### 3.3 全字段综合校验

```c
/*
 * validate_record() — 遍历所有字段逐一校验
 * 任一字段不通过立即返回对应错误码
 * 等效 SQL: 行级 CHECK 约束集合
 */
int validate_record(const Record *rec)
{
    int ret;

    if (rec == NULL) return ERR;

    ret = validate_not_empty(rec->student_id, "学号");     if (ret != OK) return ret;
    ret = validate_student_id(rec->student_id);             if (ret != OK) return ret;

    ret = validate_not_empty(rec->name, "姓名");            if (ret != OK) return ret;
    ret = validate_not_empty(rec->college, "学院");         if (ret != OK) return ret;
    ret = validate_not_empty(rec->course_id, "课程编号");   if (ret != OK) return ret;
    ret = validate_not_empty(rec->course_name, "课程名称"); if (ret != OK) return ret;

    if (rec->credit <= 0.0f || rec->credit > 20.0f)        return INVALID_SCORE;

    ret = validate_semester(rec->semester);                 if (ret != OK) return ret;
    ret = validate_enroll_date(&rec->enroll_date, rec->semester); if (ret != OK) return ret;
    ret = validate_score(rec->score);                       if (ret != OK) return ret;

    return OK;  /* 全部通过 */
}
```

## 四、完整业务层数据校验逻辑（区分普通选课与重修）

```c
/* 
 * menu_insert_record() — 用户交互插入
 * ─────────────────────────────────────
 * 流程：学号 → 课程编号 → 查重 → 其他字段 → 学期 → 日期
 *                                                 ↓
 *  普通选课：同一学生+同一课程+同一学期 → 拦截
 *  重修场景：同一学生+同一课程+不同学期 → 放行
 *           （日期校验走重修时段：目前与普通一致）
 */
static void menu_insert_record(void)
{
    Record rec;
    int duplicate;

    /* 1. 输入学号 + 格式校验 */
    while (1) {
        printf("请输入学号（12位数字）: ");
        fgets(rec.student_id, MAX_STUDENT_ID_LEN, stdin);
        rec.student_id[strcspn(rec.student_id, "\n")] = '\0';
        if (validate_student_id(rec.student_id) == OK) break;
        printf("  ⚠ %s，请重新输入\n", get_validate_error_msg(INVALID_ID));
    }

    /* 2. 输入课程编号 + 非空校验 */
    while (1) {
        printf("请输入课程编号（8位）: ");
        fgets(rec.course_id, MAX_COURSE_ID_LEN, stdin);
        rec.course_id[strcspn(rec.course_id, "\n")] = '\0';
        if (validate_not_empty(rec.course_id, "课程编号") == OK) break;
        printf("  ⚠ 课程编号不能为空，请重新输入\n");
    }

    /* ... 输入姓名、学院、课程名、学分 ... */

    /* 3. 输入学期 */
    while (1) {
        printf("请输入选课学期（YYYY-01 春季 / YYYY-02 秋季）: ");
        fgets(rec.semester, MAX_SEMESTER_LEN, stdin);
        rec.semester[strcspn(rec.semester, "\n")] = '\0';
        if (validate_semester(rec.semester) == OK) break;
        printf("  ⚠ %s，请重新输入\n", get_validate_error_msg(INVALID_SEMESTER));
    }

    /* 4. ★ 去重判断（三重键：学号+课程编号+学期） */
    duplicate = check_duplicate_triple_key(g_records, g_record_count,
                                          rec.student_id, rec.course_id, rec.semester);
    if (duplicate) {
        /* ── 普通选课：同一学期重复 → 拦截 ── */
        printf("  ❌ 错误：学号 %s 本学期（%s）已选修课程 %s，不可重复提交！\n",
               rec.student_id, rec.semester, rec.course_id);
        printf("  （如需重修请选择其他学期）\n");
        return;  /* ← 拦截，脏数据不入库 */
    }
    /* ── 不同学期 → 视为重修，自动放行 ── */

    /* 5. 输入日期 + 日期校验 */
    while (1) {
        printf("请输入选课日期（年 月 日）: ");
        fgets(input, sizeof(input), stdin);
        sscanf(input, "%d %d %d", &rec.enroll_date.year, 
               &rec.enroll_date.month, &rec.enroll_date.day);
        if (validate_enroll_date(&rec.enroll_date, rec.semester) == OK) break;
        printf("  ⚠ %s\n", get_validate_error_msg(INVALID_DATE));
    }

    /* 6. 写入数据库（数组 + 三种结构同步） */
    g_records[g_record_count] = rec;
    g_record_count++;
    list_insert(&g_list, &rec);
    avl_insert(&g_avl, &rec);
    hash_insert(&g_hash, &rec);
    printf("✅ 插入成功！当前共 %d 条记录\n", g_record_count);
}
```

## 五、异常返回提示文案一览

| 错误码 | 常量名 | 提示文案 |
|:------:|:------:|----------|
| -4 | `DUPLICATE` | ❌ **该课程本学期已完成选课，不可重复提交** |
| -5 | `INVALID_ID` | ⚠ 学号格式错误：必须为12位纯数字 |
| -6 | `INVALID_DATE` | ⚠ 选课日期错误：非法日期或与学期不匹配 |
| -7 | `INVALID_SEMESTER` | ⚠ 学期格式错误：必须为YYYY-01或YYYY-02 |
| -8 | `INVALID_SCORE` | ⚠ 成绩错误：必须在0-100之间 |
| -9 | `EMPTY_FIELD` | ⚠ 字段不能为空或纯空白 |

### 错误码定义（record.h）

```c
#define OK          0
#define ERR        -1
#define NOT_FOUND  -2
#define FULL       -3
#define DUPLICATE  -4    /* 重复选课：同一学期已选该课程 */
#define INVALID_ID -5    /* 学号格式非法 */
#define INVALID_DATE -6  /* 日期非法或超出选课时段 */
#define INVALID_SEMESTER -7  /* 学期格式错误 */
#define INVALID_SCORE  -8  /* 成绩超出0-100 */
#define EMPTY_FIELD   -9  /* 空字段 */
```

### 错误码→提示映射函数

```c
const char *get_validate_error_msg(int err_code)
{
    switch (err_code) {
        case INVALID_ID:       return "学号格式错误：必须为12位纯数字";
        case INVALID_DATE:     return "选课日期错误：非法日期或与学期不匹配";
        case INVALID_SEMESTER: return "学期格式错误：必须为YYYY-01或YYYY-02";
        case INVALID_SCORE:    return "成绩错误：必须在0-100之间";
        case EMPTY_FIELD:      return "字段不能为空或纯空白";
        case DUPLICATE:        return "该课程本学期已完成选课，不可重复提交";
        default:               return "未知错误";
    }
}
```

## 六、双重防护机制

```

───────────────                         ───────────────────

  validate_record() 校验字段               validate_record() 校验字段

  ↓                                       ↓

  合法 → 写入数组 + 同步三种结构           合法 → 加载到内存

```
